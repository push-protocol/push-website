# Example

This example demonstrates a basic wallet integration with the theme overrides for both light and dark modes.

## Live playground
```jsx live customPropMinimized = 'true'
// DO NOT FORGET TO IMPORT LIBRARIES
// NOT NEEDED HERE SINCE PLAYGROUND IMPORTS INTERNALLY
// import { PushWalletProvider, PushUniversalAccountButton, usePushWalletContext, CONSTANTS } from '@pushchain/ui-kit';

function App() {

  const walletConfig = {
    env: CONSTANTS.ENV.DEVNET,
    login: {
      email: true,
      google: true,
      wallet: {
        enabled: true,
      },
      appPreview: true,
    },
    modal: {
      loginLayout: CONSTANTS.LOGIN.SPLIT,
      connectedLayout: CONSTANTS.CONNECTED.HOVER,
      appPreview: true,
    },
  };

  const appMetadata = {
    logoUrl: "https://plus.unsplash.com/premium_photo-1746731481770-08b2f71661d0?q=80&w=2671&auto=format&fit=crop&ixlib=rb-4.1.0&ixid=M3wxMjA3fDB8MHxwaG90by1wYWdlfHx8fGVufDB8fHx8fA%3D%3D",
    title: 'Test App Title',
    description: 'Test App Description',
  };


  function WalletUI() {
    const { universalAddress } = usePushWalletContext();

    return (
      <div>
        <PushUniversalAccountButton />
      </div>
    );
  }

  return (
    <PushWalletProvider
      config={walletConfig}
      app={appMetadata}
      themeMode={CONSTANTS.THEME.LIGHT}
      themeOverrides={{
        '--pw-core-bg-primary-color': 'red',
        light: {
          '--pw-core-bg-primary-color': 'yellow',
        },
        dark: {
          '--pw-core-bg-primary-color': 'green',
        },
      }}
    >
      <WalletUI />
    </PushWalletProvider>
  );
}
```
